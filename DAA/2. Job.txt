#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

int main() {
    int n;
    cout << "Enter number of jobs: ";
    cin >> n;

    // Each job has: profit, deadline
    vector<int> profit(n), deadline(n);
    for (int i = 0; i < n; i++) {
        cout << "Enter profit and deadline for job " << i + 1 << ": ";
        cin >> profit[i] >> deadline[i];
    }

    // Create vector of (profit, deadline) pairs
    vector<pair<int, int>> jobs;
    for (int i = 0; i < n; i++) {
        jobs.push_back({profit[i], deadline[i]});
    }

    // Sort jobs by profit (descending order)
    sort(jobs.begin(), jobs.end(), [](auto &a, auto &b) {
        return a.first > b.first;
    });

    // Find max deadline to know time slots
    int maxDeadline = 0;
    for (auto &job : jobs)
        maxDeadline = max(maxDeadline, job.second);

    // Create a time slot array initialized to -1 (empty)
    vector<int> slot(maxDeadline + 1, -1);

    int totalProfit = 0, jobsDone = 0;

    // Try to schedule each job
    for (int i = 0; i < n; i++) {
        int profitNow = jobs[i].first;
        int deadlineNow = jobs[i].second;

        // Find an empty slot before or on its deadline
        for (int j = deadlineNow; j > 0; j--) {
            if (slot[j] == -1) {
                slot[j] = i; // assign job to this slot
                totalProfit += profitNow;
                jobsDone++;
                break;
            }
        }
    }

    cout << "\nTotal Jobs Done: " << jobsDone;
    cout << "\nTotal Profit: " << totalProfit << endl;

    cout << "\nOrder of Jobs (slot wise): ";
    for (int i = 1; i <= maxDeadline; i++) {
        if (slot[i] != -1)
            cout << "J" << slot[i] + 1 << " ";
        else
            cout << "- ";
    }
    cout << endl;

    return 0;
}
